$(document).ready(function() { $("#use-proxy").change(function() { if ($(this).is(":checked")) { $("#proxy").show(); } else { $("#proxy").hide(); } }); $("#rate-limit").on("input", function() { $("#rate-limit-label").text($(this).val()); }); $("#send-button").click(function() { const token = $("#token").val(); const message = $("#message").val(); const useProxy = $("#use-proxy").is(":checked"); const rateLimit = parseInt($("#rate-limit").val(), 10); const proxies = useProxy ? $("#proxy").val().split("\n").map(p => p.trim()).filter(p => p) : []; if (!token || !message) { alert("Please enter a valid token and message."); return; } $("#status").text("Sending messages..."); $("#current-proxy").text("Current Proxy: None"); sendMessages(token, message, useProxy, proxies, rateLimit); }); function sendMessages(token, message, useProxy, proxies, rateLimit) { getGuilds(token).then(guilds => { guilds.forEach((guildId, index) => { setTimeout(() => { if (useProxy) { const proxy = proxies[Math.floor(Math.random() * proxies.length)]; $("#current-proxy").text("Current Proxy: " + proxy); sendMessageWithProxy(token, message, proxy, rateLimit, guildId); } else { sendMessageToAllChannelsInGuild(token, guildId, message); } }, index * (1000 / rateLimit)); }); }); } function getGuilds(token) { return $.ajax({ url: 'https://discord.com/api/v9/users/@me/guilds', headers: { "authorization": token } }).then(response => response.map(guild => guild.id)); } function getChannels(token, guildId) { return $.ajax({ url: `https://discord.com/api/v9/guilds/${guildId}/channels`, headers: { "authorization": token } }).then(response => response.filter(channel => channel.type === 0).map(channel => channel.id)); } function sendMessageToAllChannelsInGuild(token, guildId, message) { getChannels(token, guildId).then(channels => { channels.forEach(channelId => { sendMessage(token, channelId, message); }); }); } function sendMessage(token, channelId, message) { $.ajax({ url: `https://discord.com/api/v9/channels/${channelId}/messages`, method: "POST", headers: { "authorization": token }, data: JSON.stringify({ content: message }), contentType: "application/json" }).done(() => { updateStatus(channelId); updateChannelHistory(channelId); }); } function sendMessageWithProxy(token, message, proxy, rateLimit, guildId) { getChannels(token, guildId).then(channels => { channels.forEach(channelId => { $.ajax({ url: `https://discord.com/api/v9/channels/${channelId}/messages`, method: "POST", headers: { "authorization": token }, data: JSON.stringify({ content: message }), contentType: "application/json", proxy: { http: `http://${proxy}`, https: `http://${proxy}` }, beforeSend: function(xhr) { xhr.setRequestHeader('X-FORWARDED-FOR', proxy); }, success: function() { updateStatus(channelId); updateChannelHistory(channelId); } }); }); }); } function updateStatus(channelId) { $("#status").text(`Sending message to channel ID: ${channelId}`); } function updateChannelHistory(channelId) { const currentText = $("#result").html(); const newText = `Channel ID: ${channelId}
${currentText}`; $("#result").html(newText); } });
