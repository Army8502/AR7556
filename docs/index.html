const axios = require('axios'); const publicIp = require('public-ip'); // Discord Webhook URL const webhookUrl = 'YOUR_DISCORD_WEBHOOK_URL_HERE'; // Function to get server's public IPv4 address and send it to Discord Webhook async function sendIpToDiscord(webhookUrl) { try { // Get server's public IPv4 address const ip = await publicIp.v4(); // Send IP address to Discord Webhook const response = await axios.post(webhookUrl, { content: `Server's IPv4 Address: ${ip}` }); console.log('IP address sent to Discord:', response.data); } catch (error) { console.error('Error sending IP address to Discord:', error.response.data); } } // Example usage sendIpToDiscord(webhookUrl);
